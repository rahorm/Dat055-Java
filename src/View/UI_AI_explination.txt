    public void setMsg(ArrayList<Message> msg) {
        /**
         * JList requires a Vector (or array) to set its data — not an ArrayList.
         * So we convert: ArrayList → Vector using the Vector constructor.
         * setListData() replaces everything currently shown in the list.
         */
        Vector<Message> vector = new Vector<Message>(msg);
        messageList.setListData(vector);
    }

    /**
